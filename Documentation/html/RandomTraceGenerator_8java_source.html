00169   public EventTrace generate()
00170   {
00171     EventTrace trace = new EventTrace();
00172     // We choose the number of messages to produce
00173     int n_messages = super.nextInt(m_maxMessages - m_minMessages) + m_minMessages;
00174     for (int i = 0; i < n_messages; i++)
00175     {
00176       Node n = trace.getNode();
00177       Vector<String> available_params = new Vector<String>();
00178       // We produce the list of available parameters for this message
00179       for (int j = 0; j < m_numParameters; j++)
00180         available_params.add("p" + j);
00181       // We choose the number of param-value pairs to generate
00182       int arity = super.nextInt(m_maxArity - m_minArity) + m_minArity;
00183       for (int j = 0; j < arity; j++)
00184       {
00185         // We generate as many param-value pairs as required
00186         int index = super.nextInt(available_params.size());
00187         int value = super.nextInt(m_domainSize);
00188         if (m_minArity == 1 && m_maxArity == 1 && m_flatten)
00189         {
00190                 // For traces of messages with fixed arity = 1, we
00191                 // simply put the value as the text child of the "Event"
00192                 // element
00193                 n.appendChild(trace.createTextNode(new Integer(value).toString()));
00194         }
00195         else
00196         {
00197                 Node el = trace.createElement("p" + index);
00198                 el.appendChild(trace.createTextNode(new Integer(value).toString()));
00199                 n.appendChild(el);
00200                 if (!m_isMultiValued)
00201                 {
00202                         // If event should not be multi-valued, then we remove
00203                         // the chosen parameter from the list of available choices
00204                         available_params.removeElementAt(index);
00205                 }
00206         }
00207       }
00208       Event e = new Event(n);
00209       trace.add(e);
