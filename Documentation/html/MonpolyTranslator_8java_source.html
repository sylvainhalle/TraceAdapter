00171   private StringBuilder toMonpoly(Node n, String indent)
00172   {
00173     StringBuilder out = new StringBuilder();
00174     NodeList children = n.getChildNodes();
00175     int num_children = children.getLength();
00176     if (num_children == 1 && children.item(0).getNodeType() == Node.TEXT_NODE)
00177     {
00178       Node child = children.item(0);
00179       String val = child.getNodeValue();
00180       val = val.trim();
00181       if (val.isEmpty())
00182       {
00183         out = new StringBuilder();
00184         return out;
00185       }
00186       out.append("\n\t").append(indent).append(n.getNodeName()).append("(\"")
00187           .append(val).append("\")");
00188       return out;
00189     }
00190     for (int i = 0; i < num_children; i++)
00191     {
00192       Node child = children.item(i);
00193       if (child.getNodeType() == Node.TEXT_NODE)
00194       {
00195         String val = child.getNodeValue();
00196         val = val.trim();
00197         if (val.isEmpty()) // We ignore whitespace
00198           continue;
00199       }
00200       out.append(toMonpoly(child, ""));
00201       out.append("\n");
00202     }
00203     return out;
00204   }
