00064   public String translateTrace()
00065   {
00066     StringBuilder out = new StringBuilder();
00067     out.append("<trace>\n");
00068     for (Event e : m_trace)
00069     {
00070       // If event is multi-valued, then we will lose information
00071       // by flattening it
00072       if (e.isMultiValued())
00073         m_equivalencePreserved = false;
00074       out.append("<message>\n");
00075       Relation<String,String> domain = e.getParameterDomain();
00076       Set<String> all_params = domain.keySet();
00077       for (String p : all_params)
00078       {
00079         if (!all_params.contains(p))
00080           continue;
00081         Set<String> values = domain.get(p);
00082         String val = "";
00083         // We get only the first value
00084         for (String v : values)
00085         {
00086           val = v;
00087           break;
00088         }
00089         out.append("<").append(p).append(">").append(val).append("</").append(p).append(">\n");
00090       }
00091       out.append("</message>\n");
00092     }
00093     out.append("</trace>");
00094     return out.toString();
00095   }
