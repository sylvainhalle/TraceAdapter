00077   public String translateTrace()
00078   {
00079     StringBuilder out = new StringBuilder();
00080     Relation<String,String> domains = m_trace.getParameterDomain();
00081     Set<String> params = domains.keySet();
00082     // We must convert the set into a list to force a fixed iteration order
00083     List<String> o_params = new LinkedList<String>();
00084     o_params.addAll(params);
00085     // Table contents
00086     out.append("USE ").append(m_databaseName).append(";\n");
00087     out.append("INSERT INTO ").append(m_tableName).append(" (`").append(m_eventId).append("`");
00088     for (String p_name : o_params)
00089     {
00090       out.append(",`").append(p_name).append("`");
00091     }
00092     out.append(")\nVALUES\n");
00093     for (int i = 0; i < m_trace.size(); i++)
00094     {
00095       Event e = m_trace.elementAt(i);
00096       if (!e.isFlat() || e.isMultiValued())
00097       {
00098         out.append("-- WARNING: this event is not flat or is multi-valued\n");
00099       }
00100       out.append(toSql(e, i, o_params));
00101       if (i < m_trace.size() - 1)
00102         out.append(",\n");
00103     }
00104     out.append(";");
00105     return out.toString();
00106   }
