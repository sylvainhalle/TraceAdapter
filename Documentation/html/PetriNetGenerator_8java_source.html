00222         protected void parseFromString(String s) throws java.text.ParseException
00223         {
00224                 String[] elements = s.split("\\s+");
00225                 int i = 0;
00226                 while (i < elements.length)
00227                 {
00228                         String word = elements[i];
00229                         if (word.compareTo("P") == 0)
00230                         {
00231                                 // Definition of a place -> transition arrow
00232                                 assert i + 2 < elements.length;
00233                                 String place_label = elements[i + 1];
00234                                 String trans_label = elements[i + 2];
00235                                 Place p = getAddPlace(new Place(place_label));
00236                                 Transition t = getAddTransition(new Transition(trans_label));
00237                                 p.addOutgoingTransition(t);
00238                                 t.addIncomingPlace(p);
00239                                 i += 3;
00240                         }
00241                         else if (word.compareTo("T") == 0)
00242                         {
00243                                 // Definition of a transition -> place arrow
00244                                 assert i + 2 < elements.length;
00245                                 String place_label = elements[i + 2];
00246                                 String trans_label = elements[i + 1];
00247                                 Place p = getAddPlace(new Place(place_label));
00248                                 Transition t = getAddTransition(new Transition(trans_label));
00249                                 p.addIncomingTransition(t);
00250                                 t.addOutgoingPlace(p);
00251                                 i += 3;
00252                         }
00253                         else if (word.compareTo("M") == 0)
00254                         {
00255                                 // Definition of an initial marking
00256                                 assert i + 2 < elements.length;
00257                                 String place_label = elements[i + 1];
00258                                 int value = Integer.parseInt(elements[i + 2]);
00259                                 Place p = getAddPlace(new Place(place_label));
00260                                 p.setMarking(value);
00261                                 i += 3;
00262                         }
